Usage

Now we can freely shuffle the data between PLC and the local folder.

CLASS UseManager
    VAR
        _create : BOOL;
        _read : BOOL;
        _update : BOOL;
        _delete : BOOL;
        _id : STRING;
    END_VAR   

    METHOD Use
        VAR_IN_OUT
            DataManager : AxoProcessDataManager;
        END_VAR    
        IF(_create) THEN
            IF(DataManager.Create(_id).IsDone()) THEN
                _create := FALSE;
            END_IF;
        END_IF;
        
        IF(_read) THEN
            IF(DataManager.Read(_id).IsDone()) THEN
                _read := FALSE;
            END_IF;
        END_IF;
        
        IF(_update) THEN
            IF(DataManager.Update(_id).IsDone()) THEN
                _update := FALSE;
            END_IF;
        END_IF;
        
        IF(_delete) THEN
            IF(DataManager.Delete(_id).IsDone()) THEN
                _delete := FALSE;
            END_IF;
        END_IF;
    END_METHOD        
END_CLASS
